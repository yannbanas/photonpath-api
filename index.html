quick_start.py
import requests

API_KEY = "pk_your_api_key_here"
BASE_URL = "https://photonpath-api-production.up.railway.app"

# Get optical properties of brain tissue at 630nm
response = requests.get(
    f"{BASE_URL}/v2/tissues/brain_gray_matter",
    params={"wavelength": 630},
    headers={"X-API-Key": API_KEY}
)

data = response.json()
print(f"Penetration depth: {data['derived']['penetration_depth_mm']:.2f} mm")
# ‚Üí Penetration depth: 2.82 mm
